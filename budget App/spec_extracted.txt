BudgetLens: Enterprise Technical Specification & Architecture 
Blueprint 
1. Executive Overview 
BudgetLens is an AI-native, event-driven expense intelligence platform designed for 
high-volume concurrent processing of financial documents. The system employs Clean 
Architecture with Domain-Driven Design (DDD) tactical patterns to ensure the core 
business logic remains framework-agnostic, testable, and evolvable over a 5+ year 
horizon. 
Target Scale: 100,000+ MAU, 50,000+ receipts processed daily, 99.9% uptime SLA, 
sub-200ms API response times (p95), and sub-30-second end-to-end receipt processing 
latency. 
 
2. Strategic Architectural Decisions 
2.1 Architectural Style 
●​ Primary: Clean Architecture (Robert C. Martin) with Ports & Adapters (Hexagonal 
Architecture) 
●​ Secondary: CQRS (Command Query Responsibility Segregation) for receipt 
read/write operations 
●​ Tertiary: Event Sourcing for receipt processing audit trails (optional in Phase 3)

7.2.2 OCR Worker (Consumer) 
TypeScriptCopy 
@Injectable() 
export class OcrWorker implements OnModuleInit { 
  constructor( 
    private readonly ocrStrategy: OcrStrategy, 
    private readonly receiptRepo: ReceiptRepository, 
    private readonly eventBus: EventBus, 
    private readonly metrics: MetricsService 
  ) {} 
   
  @RabbitSubscribe({ 
    exchange: 'receipts.exchange', 
    routingKey: 'receipt.uploaded', 
    queue: 'ocr-processing-queue', 
    queueOptions: { 
      durable: true, 
      arguments: { 
        'x-dead-letter-exchange': 'receipts.dlx', 
        'x-dead-letter-routing-key': 'ocr.failed' 
      } 
    } 
  }) 
  async handleReceiptUploaded(event: ReceiptUploadedEvent): Promise<void> 
{ 
    const timer = this.metrics.startTimer('ocr_processing_duration'); 
     
    try { 
      // 1. Update status to OCR_PROCESSING (optimistic locking) 
      await this.receiptRepo.updateStatus( 
        ReceiptId.create(event.aggregateId), 
        ProcessingStatus.OCR_PROCESSING, 
        /* version */ expect.any(Number) 
      ); 
       
      // 2. Download image from S3 (stream to temp file) 
      // 3. Execute OCR strategy 
      const result = await this.ocrStrategy.process(event.imageRef); 
       
      // 4. Validate confidence threshold (> 80%) 
      if (result.confidence < 0.8) { 
        // Fallback to secondary strategy or manual review queue 
      } 
       
      // 5. Emit OcrCompletedEvent 
      await this.eventBus.publish(new OcrCompletedEvent( 
        event.aggregateId, 
        result.rawText, 
        result.confidence, 
        this.ocrStrategy.name 
      )); 
       
      this.metrics.increment('ocr_success_total'); 
    } catch (error) { 
      this.metrics.increment('ocr_failure_total'); 
      // Message goes to DLQ after 3 retries 
      throw error; // Requeue for retry with backoff 
    } finally { 
      timer.end(); 
    } 
  } 
}

      response_format: { type: 'json_object' }, 
      temperature: 0.1, // Low creativity for extraction 
      max_tokens: 4096 
    }); 
     
    // Validate JSON schema using Zod

ReceiptSchema.parse(JSON.parse(response.choices[0].message.content));

@Injectable() 
export class VectorClassificationEngine { 
  constructor( 
    private readonly embeddingService: OpenAIEmbeddingService, 
    private readonly vectorStore: PineconeService, 
    private readonly categoryRepo: CategoryRepository 
  ) {} 
   
  async classify(receipt: Receipt): Promise<CategoryId> { 
    // 1. Generate embedding from merchant name + line items 
    const text = `${receipt.merchantName} ${receipt.lineItems.map(i => 
i.description).join(' ')}`; 
    const embedding = await this.embeddingService.create(text); 
     
    // 2. Semantic search against user's categories + system categories 
    const results = await this.vectorStore.query({ 
      vector: embedding, 
      filter: { 
        $or: [ 
          { userId: receipt.userId }, 
          { type: 'SYSTEM' } 
        ] 
      }, 
      topK: 3 
    }); 
     
    // 3. Apply confidence threshold 
    if (results[0].score > 0.85) { 
      return CategoryId.create(results[0].id); 
    } 
     
    // 4. Fallback to heuristic rules (regex patterns for known merchants) 
    return this.heuristicClassify(receipt); 
  } 
}

@Injectable() 
export class RateLimitService { 
  async checkLimit(userId: string, tier: SubscriptionLevel): 
Promise<boolean> { 
    const key = `ratelimit:${userId}`; 
    const limit = RATE_LIMITS[tier]; 
     
    const pipeline = this.redis.pipeline(); 
    pipeline.incr(key); 
    pipeline.expire(key, limit.window); 
    const [current] = await pipeline.exec(); 
     
    return current <= limit.requests; 
  } 
}
